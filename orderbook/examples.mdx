## Command line (wscat)

```bash
# Connect and subscribe to a market
wscat -c "wss://ob.polynode.dev/ws?key=pn_live_YOUR_KEY" \
  -x '{"action":"subscribe","markets":["73624432805780182150964443951045800666977811185963019133914618974858599458273"]}'
```

## JavaScript

```javascript
const ws = new WebSocket("wss://ob.polynode.dev/ws?key=pn_live_YOUR_KEY");

ws.onopen = () => {
  // Subscribe to Bitcoin $150K and Netanyahu markets
  ws.send(JSON.stringify({
    action: "subscribe",
    markets: [
      "73624432805780182150964443951045800666977811185963019133914618974858599458273",
      "114694726451307654528948558967898493662917070661203465131156925998487819889437",
      "66255671088804707681511323064315150986307471908131081808279119719218775249892"
    ]
  }));
};

// Maintain local orderbook state
const books = {};

ws.onmessage = (event) => {
  const msg = JSON.parse(event.data);

  switch (msg.type) {
    case "book_snapshot":
      books[msg.asset_id] = { bids: msg.bids, asks: msg.asks };
      console.log(`Snapshot: ${msg.question} [${msg.outcome}] - ${msg.bids.length}b/${msg.asks.length}a`);
      break;

    case "batch":
      for (const update of msg.updates) {
        if (update.type === "price_change") {
          for (const asset of update.assets) {
            console.log(`${update.question} [${asset.outcome}]: ${asset.price}`);
          }
        } else if (update.type === "book_update") {
          // Apply delta to local state
          const book = books[update.asset_id];
          if (book) {
            applyDelta(book.bids, update.bids);
            applyDelta(book.asks, update.asks);
          }
        }
      }
      break;

    case "subscribed":
      console.log(`Subscribed to ${msg.markets} tokens`);
      break;
  }
};

// Apply incremental orderbook updates
function applyDelta(existing, deltas) {
  for (const delta of deltas) {
    const idx = existing.findIndex(l => l.price === delta.price);
    if (delta.size === "0") {
      if (idx >= 0) existing.splice(idx, 1);
    } else if (idx >= 0) {
      existing[idx].size = delta.size;
    } else {
      existing.push(delta);
    }
  }
}

// Reconnect on disconnect
ws.onclose = () => setTimeout(() => location.reload(), 3000);
```

## Python

```python
import asyncio
import json
import websockets

async def stream_orderbook():
    url = "wss://ob.polynode.dev/ws?key=pn_live_YOUR_KEY"

    async with websockets.connect(url) as ws:
        # Subscribe to markets
        await ws.send(json.dumps({
            "action": "subscribe",
            "markets": [
                "73624432805780182150964443951045800666977811185963019133914618974858599458273",
                "114694726451307654528948558967898493662917070661203465131156925998487819889437",
            ]
        }))

        async for message in ws:
            data = json.loads(message)

            if data["type"] == "book_snapshot":
                print(f"Snapshot: {data['question']} [{data['outcome']}]")
                print(f"  Bids: {len(data['bids'])} levels")
                print(f"  Asks: {len(data['asks'])} levels")

            elif data["type"] == "batch":
                for update in data["updates"]:
                    if update["type"] == "price_change":
                        for asset in update["assets"]:
                            print(f"  {update['question']} [{asset['outcome']}]: {asset['price']}")

                    elif update["type"] == "book_update":
                        bids = len(update.get("bids", []))
                        asks = len(update.get("asks", []))
                        print(f"  Book delta: {update['question']} [{update['outcome']}] +{bids}b/+{asks}a")

            elif data["type"] == "subscribed":
                print(f"Subscribed to {data['markets']} tokens")

asyncio.run(stream_orderbook())
```

## Python with compression

```python
import asyncio
import json
import zlib
import websockets

async def stream_compressed():
    url = "wss://ob.polynode.dev/ws?key=pn_live_YOUR_KEY&compress=zlib"

    async with websockets.connect(url) as ws:
        await ws.send(json.dumps({
            "action": "subscribe",
            "markets": [
                "73624432805780182150964443951045800666977811185963019133914618974858599458273"
            ]
        }))

        async for message in ws:
            if isinstance(message, bytes):
                # Decompress binary frames
                text = zlib.decompress(message, -zlib.MAX_WBITS).decode("utf-8")
                data = json.loads(text)
            else:
                data = json.loads(message)

            print(f"[{data.get('type')}] {json.dumps(data)[:200]}")

asyncio.run(stream_compressed())
```

## Finding token IDs

To subscribe to a specific market, you need its Polymarket token IDs. You can get these from Polymarket's Gamma API:

```bash
# Search for a market by slug
curl -s "https://gamma-api.polymarket.com/events?slug=fed-decision-in-march-885" | \
  python3 -c "
import json, sys
events = json.load(sys.stdin)
for e in events:
    print(f'Event: {e[\"title\"]}')
    for m in e.get('markets', []):
        tokens = json.loads(m.get('clobTokenIds', '[]'))
        outcomes = json.loads(m.get('outcomes', '[]'))
        print(f'  {m[\"question\"]}')
        for token, outcome in zip(tokens, outcomes):
            print(f'    {outcome}: {token}')
"
```

Or search by keyword:

```bash
curl -s "https://gamma-api.polymarket.com/events/pagination?limit=5&active=true&closed=false&order=volume24hr&ascending=false" | \
  python3 -c "
import json, sys
data = json.load(sys.stdin)
for e in data.get('data', []):
    print(f'{e[\"title\"]} (vol24h: {e.get(\"volume24hr\", 0):.0f})')
    for m in e.get('markets', [])[:1]:
        tokens = json.loads(m.get('clobTokenIds', '[]'))
        print(f'  Tokens: {tokens}')
"
```
