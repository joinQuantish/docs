## Subscribe to all live games

<CodeGroup>
```javascript Node.js
import { WebSocket } from "ws";

const ws = new WebSocket("wss://sports.polynode.dev/ws?key=pn_live_YOUR_KEY");

ws.on("open", () => {
  ws.send(JSON.stringify({
    action: "subscribe",
    type: "sports_scores",
    filters: { status: ["inprogress"] }
  }));
});

ws.on("message", (raw) => {
  const msg = JSON.parse(raw.toString());
  if (msg.type === "score_update") {
    const g = msg.data;
    console.log(`[${g.leagueAbbreviation}] ${g.awayTeam} ${g.score} @ ${g.homeTeam} (${g.period})`);
  }
});

ws.on("close", () => console.log("disconnected"));
```

```python Python
import asyncio, json
import websockets

async def stream():
    url = "wss://sports.polynode.dev/ws?key=pn_live_YOUR_KEY"
    async with websockets.connect(url) as ws:
        await ws.send(json.dumps({
            "action": "subscribe",
            "type": "sports_scores",
            "filters": {"status": ["inprogress"]}
        }))
        async for raw in ws:
            msg = json.loads(raw)
            if msg.get("type") == "score_update":
                g = msg["data"]
                print(f"[{g['leagueAbbreviation']}] {g['awayTeam']} {g['score']} @ {g['homeTeam']} ({g['period']})")

asyncio.run(stream())
```

```go Go
package main

import (
    "encoding/json"
    "log"

    "github.com/gorilla/websocket"
)

func main() {
    c, _, err := websocket.DefaultDialer.Dial("wss://sports.polynode.dev/ws?key=pn_live_YOUR_KEY", nil)
    if err != nil {
        log.Fatal(err)
    }
    defer c.Close()

    sub := map[string]interface{}{
        "action":  "subscribe",
        "type":    "sports_scores",
        "filters": map[string]interface{}{"status": []string{"inprogress"}},
    }
    c.WriteJSON(sub)

    for {
        var msg map[string]interface{}
        if err := c.ReadJSON(&msg); err != nil {
            log.Println("read:", err)
            return
        }
        if msg["type"] == "score_update" {
            data := msg["data"].(map[string]interface{})
            log.Printf("[%v] %v %v @ %v (%v)\n",
                data["leagueAbbreviation"], data["awayTeam"], data["score"],
                data["homeTeam"], data["period"])
        }
    }
}
```

## Watch a single game by ID

```javascript
// First find the game ID from /games
const games = await fetch("https://sports.polynode.dev/games?sport=tennis&status=inprogress")
  .then(r => r.json());
const target = games.games[0];
console.log(`Watching game ${target.game_id}: ${target.data.awayTeam} @ ${target.data.homeTeam}`);

// Then subscribe to just that game
ws.send(JSON.stringify({
  action: "subscribe",
  type: "sports_scores",
  filters: { game_id: [target.game_id] }
}));
```

## Auto-reconnect with snapshot replay

A simple reconnect loop. The server's snapshot-on-subscribe means you'll catch up on missed events automatically as soon as you reconnect.

```javascript
import { WebSocket } from "ws";

function start() {
  const ws = new WebSocket("wss://sports.polynode.dev/ws?key=pn_live_YOUR_KEY");

  ws.on("open", () => {
    console.log("connected, subscribing");
    ws.send(JSON.stringify({
      action: "subscribe",
      type: "sports_scores",
      filters: { sport: ["tennis", "basketball"] }
      // snapshot: true is the default — replays all current matching games
    }));
  });

  ws.on("message", (raw) => {
    const msg = JSON.parse(raw.toString());
    if (msg.type === "score_update") {
      handleScore(msg.data);
    } else if (msg.type === "snapshot_complete") {
      console.log(`backlog: ${msg.events_replayed} games replayed`);
    }
  });

  ws.on("close", () => {
    console.log("disconnected, reconnecting in 2s");
    setTimeout(start, 2000);
  });

  ws.on("error", (e) => console.error("ws error:", e.message));
}

function handleScore(data) {
  console.log(`${data.awayTeam} ${data.score} @ ${data.homeTeam} (${data.period})`);
}

start();
```

## Build a live scoreboard for one tournament

```javascript
import { WebSocket } from "ws";

const ws = new WebSocket("wss://sports.polynode.dev/ws?key=pn_live_YOUR_KEY");
const board = new Map(); // gameId -> latest data

ws.on("open", () => {
  ws.send(JSON.stringify({
    action: "subscribe",
    type: "sports_scores",
    filters: { tournament: ["challenger busan"] }
  }));
});

ws.on("message", (raw) => {
  const msg = JSON.parse(raw.toString());
  if (msg.type === "score_update") {
    board.set(msg.data.gameId, msg.data);
    render();
  }
});

function render() {
  console.clear();
  console.log("Challenger Busan — Live");
  console.log("─".repeat(50));
  for (const g of board.values()) {
    console.log(`${g.awayTeam.padStart(20)} ${g.score} ${g.homeTeam.padEnd(20)} (${g.period})`);
  }
}
```

## Combine sports + settlements on different connections

If you want both Polymarket settlements AND sports scores, open two WebSocket connections in parallel.

```javascript
import { WebSocket } from "ws";

const KEY = "pn_live_YOUR_KEY";

// Settlements stream
const settlementWS = new WebSocket(`wss://ws.polynode.dev/ws?key=${KEY}`);
settlementWS.on("open", () => settlementWS.send(JSON.stringify({
  action: "subscribe", type: "settlements"
})));
settlementWS.on("message", raw => {
  const m = JSON.parse(raw);
  if (m.type === "settlement") {
    console.log(`[trade] ${m.data.taker_size} @ ${m.data.taker_price} | ${m.data.market_title}`);
  }
});

// Sports scores stream
const sportsWS = new WebSocket(`wss://sports.polynode.dev/ws?key=${KEY}`);
sportsWS.on("open", () => sportsWS.send(JSON.stringify({
  action: "subscribe", type: "sports_scores", filters: { status: ["inprogress"] }
})));
sportsWS.on("message", raw => {
  const m = JSON.parse(raw);
  if (m.type === "score_update") {
    console.log(`[score] ${m.data.awayTeam} ${m.data.score} @ ${m.data.homeTeam}`);
  }
});
```

## Detect score changes only (skip duplicate updates)

The server delivers every state update including ones where the score didn't change (e.g. clock tick, period boundary). Filter client-side if you only care about score deltas.

```javascript
const lastScore = new Map();

ws.on("message", (raw) => {
  const msg = JSON.parse(raw.toString());
  if (msg.type !== "score_update") return;
  const { gameId, score } = msg.data;
  if (lastScore.get(gameId) !== score) {
    console.log(`SCORE CHANGE on ${gameId}: ${lastScore.get(gameId) ?? "—"} → ${score}`);
    lastScore.set(gameId, score);
  }
});
```
